```python
def _check_reverse_date_anchor_coverage(self, interface_name: str, params: Dict[str, Any]) -> bool:
    """
    检查 reverse_date_range + is_date_anchor 模式的覆盖率
    复用 get_stock_existing_dates() 和 _generate_anchor_values()
    """
    ts_code = params.get("ts_code")
    start_date = params.get("start_date")
    end_date = params.get("end_date")
    
    # 获取日期锚定参数名
    interface_config = self.config_loader.get_interface_config(interface_name)
    date_anchor_param = self._find_date_anchor_param(interface_config)
    
    if ts_code:
        # 股票级检测 - 复用现有方法
        existing_dates = self.get_stock_existing_dates(interface_name, ts_code, date_anchor_param)
    else:
        # 全市场级检测 - 复用现有方法
        existing_dates = self._get_all_existing_anchors(interface_name, date_anchor_param)
    
    # 生成期望的锚点值
    expected_anchors = set(self._generate_anchor_values(start_date, end_date, date_anchor_param))
    
    # 判断是否全覆盖
    return expected_dates.issubset(existing_dates)
```

#### 3.2.2 pagination.py

**修改 `_apply_reverse_date_anchor()` 方法**：

```python
def _apply_reverse_date_anchor(self, params_stream: List[Dict[str, Any]]) -> Iterator[Dict[str, Any]]:
    """专用于 reverse_date_range + is_date_anchor 的接口"""
    for params in params_stream:
        start_date = params.get("start_date")
        end_date = params.get("end_date")
        
        if not start_date or not end_date:
            yield params
            continue
        
        date_anchor_param = self._find_date_anchor_param()
        ts_code = params.get("ts_code")
        
        # 复用 CoverageManager 获取缺失的锚点值
        if self.context.coverage_manager:
            missing_anchors = self._get_missing_anchors_via_coverage_manager(
                interface_name=self.context.interface_name,
                ts_code=ts_code,
                start_date=start_date,
                end_date=end_date,
                date_anchor_param=date_anchor_param
            )
        else:
            missing_anchors = self._generate_daily_dates(start_date, end_date)
        
        # 反向遍历
        for anchor_value in reversed(missing_anchors):
            result_params = params.copy()
            result_params[date_anchor_param] = anchor_value
            yield result_params

def _get_missing_anchors_via_coverage_manager(self, interface_name, ts_code, start_date, end_date, date_anchor_param):
    """通过 CoverageManager 获取缺失的锚点值"""
    cm = self.context.coverage_manager
    
    if ts_code:
        existing = cm.get_stock_existing_dates(interface_name, ts_code, date_anchor_param)
    else:
        existing = cm._get_all_existing_anchors(interface_name, date_anchor_param)
    
    all_anchors = set(cm._generate_anchor_values(start_date, end_date, date_anchor_param))
    return sorted(all_anchors - existing)
```

#### 3.2.3 pagination_executor.py

**精简 `_should_skip_by_coverage()` 方法**：

```python
def _should_skip_by_coverage(self, interface_config, params, coverage_manager):
    """根据覆盖率判断是否跳过请求"""
    api_name = interface_config.get('api_name', '')
    
    # 统一策略判断
    strategy = self._detect_coverage_strategy(interface_config, params)
    
    clean_params = {k: v for k, v in params.items() if not k.startswith('_')}
    
    try:
        return coverage_manager.should_skip(api_name, clean_params, strategy=strategy)
    except Exception as e:
        logger.warning(f"Coverage check failed: {e}")
        return False

def _detect_coverage_strategy(self, interface_config, params):
    """统一检测覆盖率策略"""
    pagination_config = interface_config.get('pagination', {})
    pagination_mode = pagination_config.get('mode', 'offset')
    param_defs = interface_config.get('parameters', {})
    has_date_anchor = any(p.get('is_date_anchor', False) for p in param_defs.values())
    
    if '_time_window' in params:
        return 'date_range'
    elif '_period_query' in params:
        return 'period'
    elif '_stock_info' in params:
        return 'stock'
    elif has_date_anchor and pagination_mode == 'reverse_date_range':
        return 'date_anchor_reverse'  # 新策略
    elif has_date_anchor:
        return 'period'
    else:
        return 'default'
```
