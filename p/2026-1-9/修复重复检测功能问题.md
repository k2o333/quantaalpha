# 修复重复检测功能问题

## 问题概述

当前系统中，重复检测功能未能正常工作，导致每次运行都会重新下载已存在的数据，而不是跳过已下载的记录。

## 问题根本原因

### 1. 策略不匹配
- `income_vip` 接口使用 `stock_loop` 模式
- 在 `download_single_stock` 方法中，调用覆盖率检查时使用的是 `strategy='stock'`
- 但 `_check_stock_existence` 方法检查的是 `ts_code` 字段是否已存在于数据库中

### 2. 逻辑错误
- `_check_stock_existence` 方法只是检查某个 `ts_code` 是否存在于已下载的数据中
- 但它并没有检查**特定日期范围**内该股票的数据是否已下载
- 对于 `income_vip` 接口，即使 `000001.SZ` 已经存在于数据库中，但如果日期范围不同，仍然需要下载

### 3. 配置不一致
- `income_vip.yaml` 中的 `duplicate_detection` 配置为 `key_column: period`
- 但在 `stock_loop` 模式下调用的是 `strategy='stock'`，使用 `ts_code` 作为检测字段
- 这导致了配置和实际调用的不一致

### 4. 主键检测缺失
- `income_vip` 的主键是 `["ts_code", "ann_date", "end_date"]`
- 系统没有基于完整主键组合进行重复检测

## 解决方案

### 方案一：修复 `_check_stock_existence` 方法

修改 `core/coverage_manager.py` 中的 `_check_stock_existence` 方法，使其能够检查特定日期范围内的数据：

```python
def _check_stock_existence(self, interface_name: str, params: Dict[str, Any]) -> bool:
    """
    检查股票在特定日期范围内是否存在数据
    现在会检查完整的主键组合，而不仅仅是ts_code
    """
    target_stock = params.get('ts_code')
    if not target_stock:
        logger.debug(f"Missing ts_code parameter for {interface_name}, skipping stock check")
        return False

    # 获取接口配置
    interface_config = self.config_loader.get_interface_config(interface_name)
    detection_config = interface_config.get('duplicate_detection', {})
    
    # 获取接口的主键配置
    primary_keys = interface_config.get('output', {}).get('primary_key', ['ts_code'])

    try:
        # 获取已存储数据中该股票的记录
        df = self.storage_manager.read_interface_data(
            interface_name, 
            columns=primary_keys
        )

        if df.is_empty():
            logger.debug(f"No existing data found for {interface_name}")
            return False

        # 检查是否存在相同的主键组合
        # 对于income_vip，主键是["ts_code", "ann_date", "end_date"]
        target_records = df.filter(pl.col('ts_code') == target_stock)
        
        if target_records.is_empty():
            logger.debug(f"No existing records for stock {target_stock} in {interface_name}")
            return False

        # 如果有日期参数，进一步检查日期范围
        start_date = params.get('start_date')
        end_date = params.get('end_date')
        
        if start_date and end_date:
            # 检查日期范围内的数据是否已存在
            # 对于income_vip，主要检查end_date（报告期）
            date_column = 'end_date'  # 对于财务数据，通常使用end_date作为日期字段
            if date_column in target_records.columns:
                existing_dates = set(target_records[date_column].to_list())
                
                # 获取预期的日期范围内的交易日
                calendar_params = {
                    'start_date': start_date,
                    'end_date': end_date,
                    'exchange': 'SSE'
                }
                trade_calendar = self._make_request_to_trade_cal(calendar_params)
                
                if trade_calendar:
                    expected_dates = {day['cal_date'] for day in trade_calendar if day.get('is_open', 0) == 1}
                    # 检查覆盖率
                    coverage = len(existing_dates & expected_dates) / len(expected_dates) if expected_dates else 0
                    threshold = detection_config.get('threshold', 0.95)
                    
                    result = coverage >= threshold
                    logger.info(f"Stock {target_stock} coverage in {interface_name}: {coverage:.2%} ({len(existing_dates & expected_dates)}/{len(expected_dates)})")
                    return result
                else:
                    # 如果无法获取交易日历，简单检查是否已存在该股票的任何数据
                    return not target_records.is_empty()
        else:
            # 没有日期范围限制，只要存在该股票的数据就认为已存在
            return not target_records.is_empty()

    except Exception as e:
        logger.warning(f"Stock existence check failed for {interface_name}: {e}")
        return False
```

### 方案二：增强日期范围覆盖率检查

修改 `_check_range_coverage` 方法，使其能够更精确地检测基于主键的重复数据：

```python
def _check_range_coverage(self, interface_name: str, params: Dict[str, Any]) -> bool:
    """
    检查日期范围覆盖率 - 增强版，支持基于主键的精确检测
    """
    start_date = params.get('start_date')
    end_date = params.get('end_date')

    if not start_date or not end_date:
        logger.debug(f"Missing date range parameters for {interface_name}, skipping coverage check")
        return False

    # 获取接口配置
    interface_config = self.config_loader.get_interface_config(interface_name)
    detection_config = interface_config.get('duplicate_detection', {})

    # 获取接口的主键配置
    primary_keys = interface_config.get('output', {}).get('primary_key', [])
    
    # 检查缓存
    cache_key = f"{interface_name}_{start_date}_{end_date}"
    if cache_key in self._coverage_cache:
        return self._coverage_cache[cache_key]

    try:
        # 读取接口数据，包含主键字段
        columns_to_read = primary_keys if primary_keys else ['trade_date', 'ts_code']
        
        df = self.storage_manager.read_interface_data(
            interface_name,
            start_date=start_date,
            end_date=end_date,
            columns=columns_to_read
        )

        if df.is_empty():
            logger.debug(f"No existing data found for {interface_name} in range {start_date}-{end_date}")
            self._coverage_cache[cache_key] = False
            return False

        # 检查是否指定了特定股票
        target_stock = params.get('ts_code')
        if target_stock and 'ts_code' in df.columns:
            df = df.filter(pl.col('ts_code') == target_stock)
            if df.is_empty():
                logger.debug(f"No existing data for stock {target_stock} in range {start_date}-{end_date}")
                self._coverage_cache[cache_key] = False
                return False

        # 获取实际存在的主键组合
        if len(primary_keys) > 0:
            # 基于完整主键组合进行检查
            existing_records = set()
            for row in df.iter_rows(named=True):
                key_tuple = tuple(row.get(k, None) for k in primary_keys)
                existing_records.add(key_tuple)
            
            # 获取预期的主键组合（基于日期范围）
            # 这里需要根据具体接口类型生成预期的主键组合
            # 对于财务数据，通常是基于报告期的组合
            if 'end_date' in primary_keys:
                # 获取日期范围内的交易日历
                calendar_params = {
                    'start_date': start_date,
                    'end_date': end_date,
                    'exchange': 'SSE'
                }
                trade_calendar = self._make_request_to_trade_cal(calendar_params)
                
                if trade_calendar:
                    expected_dates = {day['cal_date'] for day in trade_calendar if day.get('is_open', 0) == 1}
                    
                    # 对于财务数据，通常不是每天都有数据，而是季度末有数据
                    # 所以需要特殊处理
                    if interface_name in ['income_vip', 'balancesheet_vip', 'cashflow_vip']:
                        # 财务数据通常是季度报告，只在季度末有数据
                        expected_periods = self._get_quarter_end_dates(start_date, end_date)
                        expected_records = set()
                        for period in expected_periods:
                            if target_stock:
                                expected_records.add((target_stock, None, period))  # (ts_code, ann_date, end_date)
                            else:
                                # 如果没有指定股票，预期所有股票的该报告期
                                # 这种情况比较复杂，暂时按简单方式处理
                                pass
                    else:
                        # 对于日线数据，预期每个交易日都有数据
                        expected_records = set()
                        for date in expected_dates:
                            if target_stock:
                                expected_records.add((target_stock, date))  # (ts_code, trade_date)
                            else:
                                # 如果没有指定股票，预期所有股票的该交易日
                                pass
                    
                    # 计算覆盖率
                    if expected_records:
                        coverage = len(existing_records & expected_records) / len(expected_records)
                        threshold = detection_config.get('threshold', 0.95)
                        result = coverage >= threshold
                        logger.info(f"Coverage for {interface_name} ({start_date}-{end_date}): {coverage:.2%} ({len(existing_records & expected_records)}/{len(expected_records)})")
                        self._coverage_cache[cache_key] = result
                        return result

        # 如果没有主键信息或无法精确计算，使用原始方法
        # 获取预期的交易日历
        calendar_params = {
            'start_date': start_date,
            'end_date': end_date,
            'exchange': 'SSE'
        }
        trade_calendar = self._make_request_to_trade_cal(calendar_params)

        if not trade_calendar:
            # 如果无法获取交易日历，使用简单方法
            result = not df.is_empty()
            self._coverage_cache[cache_key] = result
            logger.info(f"Simple coverage check for {interface_name} ({start_date}-{end_date}): {'covered' if result else 'not covered'}")
            return result

        # 过滤出交易日
        expected_dates = {day['cal_date'] for day in trade_calendar if day.get('is_open', 0) == 1}
        
        if not expected_dates:
            logger.warning(f"No expected trade days in range {start_date}-{end_date}")
            self._coverage_cache[cache_key] = False
            return False

        # 获取实际存在的日期
        date_column = 'trade_date' if 'trade_date' in df.columns else 'end_date' if 'end_date' in df.columns else 'ann_date'
        if date_column in df.columns:
            actual_dates = set(df[date_column].to_list())
            coverage = len(actual_dates & expected_dates) / len(expected_dates)
            threshold = detection_config.get('threshold', 0.95)
            result = coverage >= threshold
            logger.info(f"Coverage for {interface_name} ({start_date}-{end_date}): {coverage:.2%} ({len(actual_dates & expected_dates)}/{len(expected_dates)})")
            self._coverage_cache[cache_key] = result
            return result
        else:
            # 如果没有日期列，简单检查是否有数据
            result = not df.is_empty()
            self._coverage_cache[cache_key] = result
            return result

    except Exception as e:
        logger.warning(f"Range coverage check failed for {interface_name}: {e}")
        self._coverage_cache[cache_key] = False
        return False

def _get_quarter_end_dates(self, start_date: str, end_date: str) -> List[str]:
    """
    获取日期范围内的季度末日期
    用于财务数据的预期主键生成
    """
    from datetime import datetime

    start_dt = datetime.strptime(start_date, '%Y%m%d')
    end_dt = datetime.strptime(end_date, '%Y%m%d')

    quarter_ends = [
        (3, 31),   # Q1
        (6, 30),   # Q2
        (9, 30),   # Q3
        (12, 31)   # Q4
    ]

    periods = []

    current_year = start_dt.year
    current_quarter_idx = 0

    # 找到 start_date 之后的第一个季度末
    for q_idx, (month, day) in enumerate(quarter_ends):
        quarter_end = datetime(current_year, month, day)
        if quarter_end >= start_dt:
            current_quarter_idx = q_idx
            break
    else:
        # 如果当前年份没有找到，从下一年第一季度开始
        current_year += 1
        current_quarter_idx = 0

    # 生成所有在范围内的季度末日期
    while True:
        month, day = quarter_ends[current_quarter_idx]
        quarter_end = datetime(current_year, month, day)

        if quarter_end > end_dt:
            break

        periods.append(quarter_end.strftime('%Y%m%d'))

        # 移动到下一个季度
        current_quarter_idx += 1
        if current_quarter_idx >= len(quarter_ends):
            current_quarter_idx = 0
            current_year += 1

    return periods
```

### 方案三：修改 `download_single_stock` 方法的策略选择

修改 `core/downloader.py` 中的 `download_single_stock` 方法，使其根据接口类型选择正确的策略：

```python
def download_single_stock(self, interface_config: Dict[str, Any], stock: Dict[str, Any], params: Dict[str, Any]) -> List[Dict[str, Any]]:
    """下载单只股票的数据 - 原子化方法供调度器调用"""
    try:
        stock_params = params.copy()
        stock_params['ts_code'] = stock['ts_code']

        # 设置日期范围
        if 'start_date' not in stock_params:
            # 如果没有指定起始日期，使用该股票的上市日期
            list_date = stock.get('list_date', '20050101')
            stock_params['start_date'] = list_date
        if 'end_date' not in stock_params:
            from datetime import datetime
            stock_params['end_date'] = datetime.now().strftime('%Y%m%d')

        # [修改] 根据接口类型选择合适的策略
        if self.coverage_manager:
            # 对于财务数据接口（income_vip, balancesheet_vip等），使用date_range策略
            # 因为这些接口需要检查特定日期范围内的数据
            interface_name = interface_config['api_name']
            if interface_name in ['income_vip', 'balancesheet_vip', 'cashflow_vip', 'express_vip', 
                                'fina_indicator_vip', 'fina_mainbz_vip', 'forecast_vip']:
                # 对于财务数据，使用日期范围策略
                should_skip = self.coverage_manager.should_skip(
                    interface_config['api_name'],
                    stock_params,
                    strategy='date_range'  # 使用date_range而不是stock
                )
            else:
                # 对于其他接口，使用stock策略
                should_skip = self.coverage_manager.should_skip(
                    interface_config['api_name'],
                    stock_params,
                    strategy='stock'
                )
            
            if should_skip:
                logger.info(f"Skipping stock {stock['ts_code']} for {interface_config['api_name']} (already exists)")
                return []

        logger.info(f"Downloading data for stock {stock['ts_code']}, date range: {stock_params.get('start_date')} - {stock_params.get('end_date')}")

        # 执行日期范围分页下载
        stock_data = self._execute_date_range_pagination(interface_config, stock_params)

        if stock_data:
            logger.info(f"Downloaded {len(stock_data)} records for {stock['ts_code']}")

        return stock_data or []
    except Exception as e:
        # [新增] 捕获异常，避免影响其他股票
        logger.error(f"Error downloading stock {stock['ts_code']}: {str(e)}")
        import traceback
        logger.debug(f"Full traceback: {traceback.format_exc()}")
        return []  # 返回空列表，让其他股票继续下载
```
