# App4 日期锚定接口参数传递修复方案（精确版）

## 修改目标

**仅影响**：`reverse_date_range` 模式 + `is_date_anchor: true` 的接口（目前只有 `cyq_perf`）

**不影响**：
- `stock_loop` 模式的日期锚定接口（top10_holders, stk_rewards, disclosure_date, dividend, pledge_stat）
- 所有其他普通接口

---

## 问题分析

### 当前行为（错误）

```
cyq_perf 接口（reverse_date_range + is_date_anchor=true）

用户命令: python app4/main.py --interface cyq_perf --start_date 20260205 --end_date 20260220

执行流程:
1. params_builder._detect_scenario()
   - is_stock_loop = False (mode = reverse_date_range)
   - 返回 DownloadScenario.DIRECT
   
2. pagination.compose()
   - is_date_anchor_interface = True
   - 跳过 time_range 处理
   - 但参数仍然是 {start_date: "20260205", end_date: "20260220"}
   
3. API 收到: {"start_date": "20260205", "end_date": "20260220"}
   - 报错: "参数校验失败, ts_code,trade_date至少输入一个参数"
```

### 预期行为（正确）

```
cyq_perf 接口（reverse_date_range + is_date_anchor=true）

用户命令: python app4/main.py --interface cyq_perf --start_date 20260205 --end_date 20260220

执行流程:
1. params_builder._detect_scenario()
   - 检测到 is_date_anchor=true 且 mode=reverse_date_range
   - 返回 DownloadScenario.DATE_ANCHOR_RANGE（新场景）
   
2. pagination.compose()
   - 将 start_date/end_date 转换为单个 trade_date 值
   - 生成多个请求: {trade_date: "20260205"}, {trade_date: "20260206"}, ...
   
3. API 收到多个请求，每个请求格式正确
   - 成功返回数据
```

```python
def _build_date_anchor_range_params(
    self,
    start_date: str,
    end_date: Optional[str],
    ts_code: Optional[str]
) -> BuildResult:
    """
    构建日期锚定范围参数
    
    专用于 reverse_date_range + is_date_anchor 的接口（如 cyq_perf）
    将用户提供的日期范围存储为内部参数，后续转换为单个日期锚定值
    """
    params: Dict[str, Any] = {}
    date_anchor_param = self._find_date_anchor_param()
    
    # 使用内部参数存储日期范围，避免与 API 参数冲突
    if start_date:
        params['_date_anchor_start'] = start_date
    if end_date:
        params['_date_anchor_end'] = end_date
    if ts_code:
        params['ts_code'] = ts_code
    
    return BuildResult(
        params=params,
        scenario=DownloadScenario.DATE_ANCHOR_RANGE,
        requires_stock_loop=False,
        date_anchor_param=date_anchor_param
    )
```

```python
def _build_date_anchor_range_params_list(self, result: BuildResult) -> List[Dict[str, Any]]:
    """
    构建日期锚定范围参数列表
    
    将日期范围转换为单个日期锚定值的列表
    每个请求只包含一个日期锚定值，不带 ts_code（API 会返回该日所有股票数据）
    """
    start_date = result.params.get('_date_anchor_start')
    end_date = result.params.get('_date_anchor_end')
    date_anchor_param = result.date_anchor_param
    
    if not start_date or not date_anchor_param:
        return []
    
    # 生成日期锚定值列表
    anchor_values = self._generate_daily_anchor_values(start_date, end_date)
    
    params_list = []
    for anchor_value in anchor_values:
        p = {date_anchor_param: anchor_value}
        # 如果用户指定了 ts_code，也传递
        if result.params.get('ts_code'):
            p['ts_code'] = result.params['ts_code']
        params_list.append(p)
    
    return params_list

def _generate_daily_anchor_values(self, start_date: str, end_date: Optional[str]) -> List[str]:
    """
    生成每日日期锚定值
    
    Args:
        start_date: 开始日期 YYYYMMDD
        end_date: 结束日期 YYYYMMDD
        
    Returns:
        日期字符串列表，如 ['20260205', '20260206', '20260207']
    """
    if not start_date:
        return []
    
    try:
        start = datetime.strptime(start_date, '%Y%m%d')
        end = datetime.strptime(end_date, '%Y%m%d') if end_date else datetime.now()
    except ValueError:
        return []
    
    dates = []
    current = start
    while current <= end:
        dates.append(current.strftime('%Y%m%d'))
        current += timedelta(days=1)
    
    return dates
```

### 修改点 8: pagination.py - 新增日期锚定范围处理方法

**文件**: `app4/core/pagination.py`

**位置**: 在 `_apply_period_range` 方法后（约第 186 行后）

**修改内容**: 新增 `_apply_date_anchor_range` 方法

```python
def _apply_date_anchor_range(
    self, params_stream: List[Dict[str, Any]]
) -> Iterator[Dict[str, Any]]:
    """
    应用日期锚定范围维度
    
    专用于 reverse_date_range + is_date_anchor 的接口（如 cyq_perf）
    将内部参数 _date_anchor_start/_date_anchor_end 转换为单个日期锚定值
    
    Args:
        params_stream: 参数流
        
    Yields:
        应用日期锚定后的参数
    """
    date_anchor_param = self._find_date_anchor_param()
    if not date_anchor_param:
        yield from params_stream
        return
    
    for params in params_stream:
        start_date = params.pop('_date_anchor_start', params.get('start_date'))
        end_date = params.pop('_date_anchor_end', params.get('end_date'))
        
        # 移除原始的日期范围参数
        params.pop('start_date', None)
        params.pop('end_date', None)
        
        if not start_date:
            yield params
            continue
        
        # 生成每日日期锚定值
        anchor_values = self._generate_daily_dates(start_date, end_date)
        
        # 反向遍历（reverse_date_range 模式）
        anchor_values = list(reversed(anchor_values))
        
        for anchor_value in anchor_values:
            p = params.copy()
            p[date_anchor_param] = anchor_value
            p['_date_anchor_value'] = anchor_value  # 内部标记
            yield p

def _find_date_anchor_param(self) -> Optional[str]:
    """
    查找日期锚定参数名称
    
    Returns:
        日期锚定参数名称，如 'trade_date'、'ann_date' 等
    """
    parameters = self.interface_config.get("parameters", {})
    for param_name, param_def in parameters.items():
        if param_def.get("is_date_anchor", False):
            return param_name
    return None

def _generate_daily_dates(self, start_date: str, end_date: Optional[str]) -> List[str]:
    """
    生成每日日期列表
    
    Args:
        start_date: 开始日期 YYYYMMDD
        end_date: 结束日期 YYYYMMDD
        
    Returns:
        日期字符串列表
    """
    if not start_date:
        return []
    
    try:
        start = datetime.strptime(start_date, '%Y%m%d')
        end = datetime.strptime(end_date, '%Y%m%d') if end_date else datetime.now()
    except ValueError:
        return []
    
    dates = []
    current = start
    while current <= end:
        dates.append(current.strftime('%Y%m%d'))
        current += timedelta(days=1)
    
    return dates
```
