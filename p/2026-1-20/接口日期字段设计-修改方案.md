### 修改1：TradeDateCache 改为模块级变量

**原文：**
```python
class TradeDateCache:
    """交易日历内存缓存 - 单例模式"""
    
    _instance = None
    _lock = threading.Lock()
    _date_map = None
    _is_open_map = None
    
    @classmethod
    def get_cache(cls):
        """获取缓存实例 - 惰性加载"""
        if cls._date_map is None:
            with cls._lock:
                if cls._date_map is None:  # 双重检查
                    cls._load_cache()
        return cls._date_map
```

**修改为：**
```python
# 模块级变量，多进程下各自独立，但不需要锁
_date_map = None
_is_open_map = None
_cache_loaded = False
_cache_lock = threading.Lock()

def get_trade_date_cache():
    """获取交易日历缓存 - 模块级单例"""
    global _date_map, _is_open_map, _cache_loaded
    
    if not _cache_loaded:
        with _cache_lock:
            if not _cache_loaded:
                _load_cache()
                _cache_loaded = True
    
    return _date_map, _is_open_map

def _load_cache():
    """一次性加载trade_cal到内存"""
    global _date_map, _is_open_map
    
    # 从 data 目录加载
    from storage import StorageManager
    storage = StorageManager()
    
    # 读取parquet文件
    import polars as pl
    df = pl.read_parquet(f"{storage.base_dir}/trade_cal/*.parquet")
    
    # 只加载需要的字段
    _date_map = {
        row['cal_date']: row['cal_date_dt'] 
        for row in df.to_dicts()
    }
    
    _is_open_map = {
        row['cal_date']: row['is_open_bool'] 
        for row in df.to_dicts()
    }
    
    print(f"TradeDateCache loaded: {len(_date_map)} dates")
```

**使用示例：**
```python
def convert_trade_date(date_str: str) -> date:
    """转换交易日"""
    if not date_str:
        return None
    
    date_map, _ = get_trade_date_cache()
    result = date_map.get(date_str)
    
    if result is None:
        # 降级方案：直接解析
        logger.warning(f"日期 {date_str} 不在trade_cal中，直接解析")
        result = datetime.strptime(date_str, '%Y%m%d').date()
    
    return result
```

def process_data(self, data: List[Dict[str, Any]], interface_config: Dict[str, Any]) -> pl.DataFrame:
    if not data:
        return pl.DataFrame()

    try:
        interface_name = interface_config.get('api_name', 'unknown')

        # ========== 步骤1：创建DataFrame（不包含derived_fields） ==========
        df = SchemaManager.create_dataframe_without_derived(data, interface_name)

        df = self._remove_duplicates(df, interface_config)

        # ========== 步骤6：数据清洗 ==========

**新增方法：**
```python
def _filter_null_primary_keys(self, df: pl.DataFrame, interface_config: Dict[str, Any]) -> pl.DataFrame:
    """删除主键字段包含空值的记录"""
    output_config = interface_config.get('output', {})
    primary_keys = output_config.get('primary_key', [])

    # 只保留在数据中存在的主键字段
    existing_keys = [key for key in primary_keys if key in df.columns]

    if existing_keys:
        # 构建过滤条件：所有主键字段都不为空
        filter_expr = pl.all([pl.col(key).is_not_null() for key in existing_keys])
        df = df.filter(filter_expr)
        
        removed_count = len(df) if 'before_filter' in dir() else 0  # 需要记录原始数量
        logger.debug(f"Filtered {removed_count} records with null primary keys")

    return df
```

**SchemaManager 新增方法：**
```python
@staticmethod
def create_dataframe_without_derived(data: List[Dict[str, Any]], interface_name: str) -> pl.DataFrame:
    """创建DataFrame - 不应用derived_fields"""
    if not data:
        return pl.DataFrame()

    # 直接创建DataFrame
    df = pl.DataFrame(data, infer_schema_length=min(len(data), 100))
    
    # 确保列名是字符串类型
    df.columns = [str(col) for col in df.columns]
    
    # 移除重复列名
    df = df.select([col for i, col in enumerate(df.columns) if col not in df.columns[:i]])

    return df
```

### 修改4：trade_cal 自检逻辑移到下载阶段

**原文：**
```python
# 自检时机：
# - 应用启动时（预加载trade_cal）
# - 首次调用日期转换功能前
```

**修改为：**
```python
# 自检时机：
# - trade_cal 接口下载完成后立即执行
# - 系统启动时验证本地 trade_cal 数据完整性
```

**新增 - Downloader 中的自检逻辑：**
```python
class GenericDownloader:
    def __init__(self, config_loader, storage_manager, cache_manager):
        self.config_loader = config_loader
        self.storage_manager = storage_manager
        self.cache_manager = cache_manager
    
    def _verify_trade_calendar_integrity(self, df: pl.DataFrame) -> bool:
        """验证交易日历完整性"""
        from datetime import date
        import logging
        logger = logging.getLogger(__name__)
        
        # 检查1：从1990-01-01到今天是否全覆盖
        expected_start = date(1990, 1, 1)
        expected_end = date.today()
        
        actual_start = df['cal_date'].min()
        actual_end = df['cal_date'].max()
        
        if actual_start is None or actual_end is None:
            logger.error("Trade calendar data is empty")
            return False
        
        if actual_start > expected_start or actual_end < expected_end:
            logger.error(f"Trade calendar data incomplete: "
                        f"expected {expected_start} to {expected_end}, "
                        f"got {actual_start} to {actual_end}")
            return False
        
        # 检查2：string格式和date格式是否一一对应
        mismatched = df.filter(
            pl.col('cal_date').is_not_null() & pl.col('cal_date_dt').is_null()
        )
        
        if len(mismatched) > 0:
            logger.error(f"Found {len(mismatched)} records with mismatched date formats")
            return False
        
        # 检查3：数据量合理性（至少应该有8000个交易日）
        if len(df) < 8000:
            logger.warning(f"Trade calendar has unusually few records: {len(df)}")
        
        logger.info(f"Trade calendar integrity check passed: {len(df)} records")
        return True
    
    def _after_download(self, interface_name: str, df: pl.DataFrame):
        """下载后处理 - 特定接口的特殊处理"""
        if interface_name == 'trade_cal':
            if not self._verify_trade_calendar_integrity(df):
                logger.error("Trade calendar integrity check failed, will retry on next run")
                # 标记需要重新下载
                self.cache_manager.mark_dirty('trade_cal')
```
