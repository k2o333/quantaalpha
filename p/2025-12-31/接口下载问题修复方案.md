#### 2.1.3 修复数据截断问题

```python
def fetch_with_pagination(interface_func, max_records=2000, **params):
    """
    分页获取数据，避免API限制导致的截断
    """
    import pandas as pd
    from datetime import datetime, timedelta
    
    # 如果有日期范围参数，按小时间窗口分页
    if 'start_date' in params and 'end_date' in params:
        start = datetime.strptime(params['start_date'], '%Y%m%d')
        end = datetime.strptime(params['end_date'], '%Y%m%d')
        
        # 按30天为一个窗口分页
        window_size = 30
        current = start
        all_data = []
        
        while current <= end:
            window_end = min(current + timedelta(days=window_size), end)
            
            # 更新参数
            page_params = params.copy()
            page_params['start_date'] = current.strftime('%Y%m%d')
            page_params['end_date'] = window_end.strftime('%Y%m%d')
            
            try:
                data = interface_func(**page_params)
                if not data.empty:
                    all_data.append(data)
            except Exception as e:
                print(f"Error fetching data for {current} to {window_end}: {e}")
            
            current = window_end + timedelta(days=1)
        
        return pd.concat(all_data, ignore_index=True) if all_data else pd.DataFrame()
    else:
        # 如果没有日期参数，直接调用
        return interface_func(**params)
```

#### 2.2.2 改进缓存策略

**修复方案**:
```python
import hashlib
import os
from datetime import datetime, timedelta

class DataCache:
    def __init__(self, cache_dir='/home/quan/testdata/aspipe_v4/cache'):
        self.cache_dir = cache_dir
        os.makedirs(cache_dir, exist_ok=True)
    
    def get_cache_key(self, interface_name, **params):
        """生成缓存键"""
        param_str = str(sorted(params.items()))
        key_str = f"{interface_name}_{param_str}"
        return hashlib.md5(key_str.encode()).hexdigest()
    
    def get_cached_data(self, cache_key, max_age_hours=24):
        """获取缓存数据"""
        cache_file = os.path.join(self.cache_dir, f"{cache_key}.pkl")
        
        if os.path.exists(cache_file):
            import pickle
            from datetime import datetime
            
            # 检查文件修改时间
            mod_time = datetime.fromtimestamp(os.path.getmtime(cache_file))
            if (datetime.now() - mod_time).total_seconds() < max_age_hours * 3600:
                with open(cache_file, 'rb') as f:
                    return pickle.load(f)
        
        return None
    
    def cache_data(self, cache_key, data):
        """缓存数据"""
        import pickle
        cache_file = os.path.join(self.cache_dir, f"{cache_key}.pkl")
        with open(cache_file, 'wb') as f:
            pickle.dump(data, f)

def cached_fetch(interface_func, cache_enabled=True, max_age_hours=24, **params):
    """
    带缓存的数据获取
    """
    if not cache_enabled:
        return interface_func(**params)
    
    # 生成缓存键
    cache = DataCache()
    cache_key = cache.get_cache_key(interface_func.__name__, **params)
    
    # 尝试从缓存获取
    cached_data = cache.get_cached_data(cache_key, max_age_hours)
    if cached_data is not None:
        return cached_data
    
    # 从API获取数据
    data = interface_func(**params)
    
    # 缓存数据
    cache.cache_data(cache_key, data)
    
    return data
```

#### 2.2.1 优化重复数据处理

**问题**: 多个接口存在大量重复数据（如cyq_chips从6000条去重到57条）

**根本原因分析**:
1. API可能返回了重复的历史数据
2. 分页获取时时间窗口重叠
3. 缓存机制未正确处理时间范围

**增强修复方案**:
```python
class SmartDedupEngine:
    """智能去重引擎"""

    def __init__(self):
        self.dedup_strategies = {
            'cyq_chips': self.dedup_by_ts_code_trade_date,
            'cyq_perf': self.dedup_by_ts_code_trade_date,
            'daily': self.dedup_by_ts_code_trade_date,
            'daily_basic': self.dedup_by_ts_code_trade_date,
            'moneyflow': self.dedup_by_ts_code_trade_date,
            'default': self.dedup_by_all_columns
        }

    def dedup_by_ts_code_trade_date(self, data):
        """按股票代码和交易日期去重"""
        if data.empty:
            return data

        # 确保有必要的列
        required_cols = ['ts_code', 'trade_date']
        if not all(col in data.columns for col in required_cols):
            return self.dedup_by_all_columns(data)

        # 按交易日期排序，保留最新数据
        data_sorted = data.sort_values(['ts_code', 'trade_date'])
        return data_sorted.drop_duplicates(
            subset=['ts_code', 'trade_date'],
            keep='last'
        ).reset_index(drop=True)

    def dedup_by_all_columns(self, data):
        """按所有列去重"""
        return data.drop_duplicates().reset_index(drop=True)

    def smart_dedup(self, data, interface_name):
        """智能去重"""
        strategy = self.dedup_strategies.get(interface_name, self.dedup_strategies['default'])

        before_count = len(data)
        result = strategy(data)
        after_count = len(result)

        if before_count != after_count:
            print(f"去重效果: {interface_name} - {before_count} -> {after_count} (去除 {before_count - after_count} 条)")

        return result
```

    def save_range_cache(self):

    def get_missing_ranges(self, interface_name, start_date, end_date):
        """获取需要下载的缺失日期范围"""
        cache_key = f"{interface_name}_{start_date}_{end_date}"

        # 检查是否已有完整缓存
        if cache_key in self.range_cache:

    def update_cache_record(self, interface_name, start_date, end_date):
        """更新缓存记录"""
        cache_key = f"{interface_name}_{start_date}_{end_date}"
        self.range_cache[cache_key] = {'start': start_date, 'end': end_date}
        self.save_range_cache()

# 集成到数据获取流程中
def smart_fetch_with_cache(interface_name, interface_func, start_date, end_date, **params):
    """智能缓存获取"""
    cache = DateRangeCache()

    # 获取需要下载的缺失范围
    missing_ranges = cache.get_missing_ranges(interface_name, start_date, end_date)

    all_data = []

        data = interface_func(**range_params)
