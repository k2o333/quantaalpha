# 接口错误处理优化方案

## 问题描述
当前系统在遇到API访问限制错误时（如"抱歉，您每天最多访问该接口100000次"），无法正确识别和处理，导致系统在达到API限制后继续发送大量请求，产生大量重复错误日志。

## 目标
实现"遇到网络请求导致的error 3次以上，就停止这一接口的下载"的功能。

## 解决方案

### 1. 修改错误计数机制
在 `TuShareDownloader` 类中添加错误计数功能：

```python
class TuShareDownloader:
    def __init__(self):
        # ... 现有代码 ...

        # 错误计数器，用于跟踪每个接口的连续错误次数
        self.error_counts = {}
        # 接口禁用状态，达到错误次数上限后禁用接口
        self.disabled_apis = set()

    def increment_error_count(self, api_name: str) -> bool:
        """增加错误计数，如果达到限制则禁用接口"""
        current_count = self.error_counts.get(api_name, 0)
        current_count += 1
        self.error_counts[api_name] = current_count

        if current_count >= 3:
            self.disabled_apis.add(api_name)
            self.logger.warning(f"接口 {api_name} 达到连续错误次数上限，已禁用")
            return True  # 接口应被禁用
        else:
            self.logger.warning(f"接口 {api_name} 连续错误次数: {current_count}/3")
            return False  # 接口未被禁用

    def reset_error_count(self, api_name: str):
        """重置错误计数"""
        if api_name in self.error_counts:
            del self.error_counts[api_name]
        if api_name in self.disabled_apis:
            self.disabled_apis.remove(api_name)
```

### 2. 修改 `download_with_pagination` 方法中的错误处理

在 `download_with_pagination` 方法中，修改错误处理逻辑：

```python
def download_with_pagination(self, api_func, limit_per_call=2000, **base_kwargs):
    all_data = []
    offset = 0
    max_retries = 3  # 仅对网络错误重试
    retry_count = 0

    api_name = getattr(api_func, '__name__', 'unknown_api')

    # 检查接口是否被禁用
    if api_name in self.disabled_apis:
        self.logger.warning(f"接口 {api_name} 已被禁用，跳过下载")
        return pd.DataFrame()

    while True:
        # 检查接口是否被禁用（再次检查，以防在循环过程中被禁用）
        if api_name in self.disabled_apis:
            self.logger.warning(f"接口 {api_name} 已被禁用，退出下载循环")
            break

        # 添加分页参数
        kwargs = base_kwargs.copy()
        kwargs['offset'] = offset
        kwargs['limit'] = limit_per_call

        try:
            # 实现API调用前的频率限制
            self._rate_limit(api_name)
            data = api_func(**kwargs)
            retry_count = 0  # 成功则重置重试计数
            self.reset_error_count(api_name)  # 成功调用后重置错误计数
        except Exception as e:
            error_msg = str(e).lower()
            self.logger.error(f"分页下载失败, offset={offset}: {e}")

            # 检查错误是否是访问限制错误
            if self._is_access_limit_error(str(e)):
                # 访问限制错误，增加错误计数
                should_disable = self.increment_error_count(api_name)
                if should_disable:
                    break  # 达到限制后停止下载
                else:
                    # 虽然增加了错误计数，但还未达到限制，继续尝试
                    self.logger.warning(f"API访问限制错误，但未达到禁用阈值，继续尝试")
                    break  # 访问限制通常应停止分页下载
            elif "指定数据不存在" in str(e) or "data does not exist" in error_msg:
                # 数据不存在错误，不应该重试，直接结束
                self.logger.info(f"数据不存在，结束下载: {e}")
                break
            elif any(keyword in error_msg for keyword in ["timeout", "connection", "tushare.xyz", "network"]):
                # 网络相关错误，可以重试
                retry_count += 1
                if retry_count <= max_retries:
                    wait_time = min(10 * retry_count, 60)  # 指数退避，最大60秒
                    self.logger.warning(f"网络错误，等待 {wait_time:.2f} 秒后重试 (第{retry_count}/{max_retries}次)")
                    time.sleep(wait_time)
                    continue
                else:
                    self.logger.error(f"网络错误超过最大重试次数，跳过当前分页")
                    break
            else:
                # 其他错误，增加错误计数并决定是否重试
                should_disable = self.increment_error_count(api_name)
                if should_disable:
                    break  # 达到限制后停止下载
                else:
                    self.logger.error(f"非网络错误，停止下载: {e}")
                    break

        if data is None or len(data) == 0:
            break

        # 将DataFrame添加到列表中
        all_data.append(data)

        # 如果返回数据少于限制数量，说明已到最后一页
        if len(data) < limit_per_call:
            break

        # 更新offset
        offset += limit_per_call

        # 在每次API调用之间添加延迟
        time.sleep(random.uniform(0.5, 1.0))

    # 合并所有数据
    if all_data:
        return pd.concat(all_data, ignore_index=True)
    else:
        return pd.DataFrame()

def _is_access_limit_error(self, error_msg: str) -> bool:
    """检查错误信息是否是访问限制错误"""
    access_limit_keywords = [
        "每天最多访问该接口",
        "访问限制",
        "超出访问限制",
        "daily limit",
        "访问次数超限",
        "频次超限"
    ]

    for keyword in access_limit_keywords:
        if keyword in error_msg:
            return True
    return False
```

### 3. 修改 `download_with_retry` 方法以支持接口禁用检测

```python
@retry_on_failure(max_retries=3, delay=1.0, backoff=2.0)
def download_with_retry(self, api_func, *args, max_retries: int = 3, **kwargs):
    api_name = api_func.__name__ if hasattr(api_func, '__name__') else 'unknown_api'

    # 检查接口是否已被禁用
    if api_name in self.disabled_apis:
        self.logger.warning(f"接口 {api_name} 已被禁用，跳过下载")
        return None

    for attempt in range(max_retries + 1):
        try:
            # Implement rate limiting
            self._rate_limit(api_name)

            # Log the API call
            self.logger.info(f"Calling {api_name} API attempt {attempt + 1}")

            # Make the API call
            result = api_func(*args, **kwargs)

            # 调用成功，重置错误计数
            self.reset_error_count(api_name)

            self.logger.info(f"Successfully called {api_name}, got {len(result) if hasattr(result, '__len__') else 'unknown'} records")
            return result

        except Exception as e:
            error_msg = str(e).lower()
            self.logger.warning(f"Attempt {attempt + 1} failed for {api_name}: {str(e)}")

            # 检查是否是API访问限制错误
            if self._is_access_limit_error(str(e)):
                should_disable = self.increment_error_count(api_name)
                if should_disable:
                    self.logger.error(f"接口 {api_name} 达到连续错误次数上限，停止重试")
                    return None

            # Check if the error is related to token authentication
            if "token" in error_msg or "auth" in error_msg:
                # Try switching to the other token
                if self.primary_token and self.secondary_token:
                    self.switch_token()
                    self.logger.info(f"Switched token due to authentication error. Retrying {api_name}...")
                    # Retry immediately with the new token
                    try:
                        result = api_func(*args, **kwargs)
                        # 调用成功，重置错误计数
                        self.reset_error_count(api_name)
                        self.logger.info(f"Successfully called {api_name} after token switch, got {len(result) if hasattr(result, '__len__') else 'unknown'} records")
                        return result
                    except Exception as retry_e:
                        self.logger.warning(f"Retry with switched token failed for {api_name}: {str(retry_e)}")

            if attempt == max_retries:
                self.logger.error(f"All {max_retries + 1} attempts failed for {api_name}")

                # 检查是否是访问限制错误以决定是否禁用接口
                if self._is_access_limit_error(str(e)):
                    self.increment_error_count(api_name)

                ErrorHandler.handle_api_error(e, f"API call {api_name}")

            # Exponential backoff: wait longer between each attempt
            wait_time = 2 ** attempt
            self.logger.info(f"Waiting {wait_time}s before next attempt...")
            time.sleep(wait_time)
```

### 4. 修改错误处理器以识别访问限制错误

在 `error_handler.py` 中扩展错误处理：

```python
def handle_api_error(error: Exception, context: str = ""):
    """
    增强的API错误处理，区分错误类型
    """
    error_msg = str(error).lower()

    # 扩展对访问限制错误的识别
    if ("limit" in error_msg or "频次" in error_msg or "frequency" in error_msg
        or "每天最多访问该接口" in str(error) or "访问限制" in str(error)):
        logger.error(f"API频率限制超出 in {context}: {error}")
        time.sleep(120)  # 频率限制等待更长时间
    elif "token" in error_msg or "auth" in error_msg:
        logger.error(f"认证错误 in {context}: {error}")
        raise error
    elif any(keyword in error_msg for keyword in ["network", "timeout", "connection", "tushare.xyz"]):
        logger.warning(f"网络错误 in {context}: {error}")
        time.sleep(30)  # 网络错误等待30秒后重试
    elif "指定数据不存在" in str(error):
        logger.warning(f"数据不存在 in {context}: {error}")
        # 数据不存在错误不需要抛出异常，由调用方处理
        pass
    else:
        logger.error(f"未知错误 in {context}: {error}")

    # 对于数据不存在错误，不抛出异常
    if "指定数据不存在" not in str(error):
        raise error
```
