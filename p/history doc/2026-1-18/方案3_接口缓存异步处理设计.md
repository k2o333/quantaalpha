## 一、方案概述

### 核心设计
每个接口拥有独立的内存缓存，累积数据达到5000条时触发异步处理流程（去重、验证、保存），成功后清空缓存继续累积。

### 架构优势
- ✅ **内存速度快**：数据在内存中累积，减少IO等待
- ✅ **接口隔离**：每个接口独立缓存，互不干扰
- ✅ **批量处理**：5000条触发一次，平衡内存和性能
- ✅ **异步不阻塞**：下载线程继续工作，处理线程独立运行
- ✅ **失败重试**：3次重试机制，失败停止接口并通知

```
│              Data Queue → Writer Thread → Parquet           │
│                                                             │
│  data_queue: Queue                                          │

│  → 最终写入Parquet文件                                     │
└─────────────────────────────────────────────────────────────┘
```

### 2.2 数据流向

```
API下载 → Downloader → StorageManager.add_to_buffer()
                                   ↓
                          累积到5000条
                                   ↓
                          复制数据 → 清空缓存
                                   ↓
                          process_queue.put()
                                   ↓
                          _process_worker() (独立线程)
                                   ↓
                          processor.process_data() (内存去重)
                                   ↓
                          读取Parquet (跨批次去重)
                                   ↓
                          storage_manager.save_data(async=True)
                                   ↓
                          data_queue → _writer_worker() → Parquet
```

### 2.3 线程模型

**主线程**（Downloader）：
- 下载数据 → add_to_buffer() → 继续下载（不等待）
- 检查 failed_interfaces，如果接口失败则停止

**处理线程**（process_thread）：
- 从process_queue取数据 → 处理 → 放入data_queue
- 失败重试，3次失败后加入failed_interfaces

**写入线程**（writer_thread）：
- 从data_queue取数据 → 写入Parquet文件

```python
class StorageManager:
    """存储管理器 - 支持接口缓存和异步处理"""

    def __init__(self, processor: 'DataProcessor', storage_dir: str = "../data",
                 format: str = "parquet", batch_size: int = 100):
        # 现有属性
        self.storage_dir = storage_dir
        self.format = format
        self.batch_size = batch_size
        self.data_queue = queue.Queue()
        self.writer_thread = None
        self.running = False

        # 新增属性
        self.processor = processor  # 持有Processor引用
        self.interface_buffers = {}  # 接口缓存 {interface_name: BufferContext}
        self.process_queue = queue.Queue()  # 处理队列
        self.process_thread = None  # 处理线程
        self.buffer_threshold = 5000  # 触发阈值
        self.buffer_lock = threading.Lock()  # 缓存锁
        self.failed_interfaces = set()  # 失败接口集合

        # 确保存储目录存在
        os.makedirs(storage_dir, exist_ok=True)

    def start_writer(self):
        """启动写入线程和处理线程"""
        if not self.running:
            self.running = True

            # 启动写入线程（现有）
            self.writer_thread = threading.Thread(
                target=self._writer_worker,
                daemon=True
            )
            self.writer_thread.start()

            # 启动处理线程（新增）
            self.process_thread = threading.Thread(
                target=self._process_worker,
                daemon=True
            )
            self.process_thread.start()

            logger.info("Storage writer and process threads started")

    def stop_writer(self):
        """停止所有线程"""
        if self.running:
            self.running = False

            # 停止处理线程
            self.process_queue.put(None)  # 发送哨兵
            if self.process_thread:
                self.process_thread.join()

            # 停止写入线程
            self.data_queue.put(None)  # 发送哨兵
            if self.writer_thread:
                self.writer_thread.join()

            logger.info("Storage threads stopped")

    def add_to_buffer(self, interface_name: str, data: List[Dict[str, Any]]):
        """
        添加数据到接口缓存

        Args:
            interface_name: 接口名称
            data: 数据列表
        """
        with self.buffer_lock:
            # 初始化接口缓存（如果不存在）
            if interface_name not in self.interface_buffers:
                self.interface_buffers[interface_name] = {
                    'data': [],
                    'count': 0
                }

            buffer = self.interface_buffers[interface_name]

            # 累积数据
            buffer['data'].extend(data)
            buffer['count'] += len(data)

            # 检查是否达到阈值
            if buffer['count'] >= self.buffer_threshold:
                # 复制数据用于处理
                data_to_process = buffer['data'].copy()

                # 重置缓存
                buffer['data'] = []
                buffer['count'] = 0

                # 放入处理队列
                try:
                    self.process_queue.put({
                        'interface_name': interface_name,
                        'data': data_to_process,
                        'retry_count': 0,
                        'timestamp': time.time()
                    }, block=False)
                    logger.debug(f"Queued {len(data_to_process)} records for processing: {interface_name}")
                except queue.Full:
                    logger.error(f"Process queue is full, dropping data for {interface_name}")

    def _process_worker(self):
        """处理线程：数据去重、验证、放入写入队列"""
        while True:
            try:
                task = self.process_queue.get(timeout=1)

                # 检查停止信号
                if task is None:
                    logger.info("Process worker received stop signal")
                    break

                # 检查接口是否已失败
                interface_name = task['interface_name']
                if interface_name in self.failed_interfaces:
                    logger.warning(f"Skipping processing for failed interface: {interface_name}")
                    continue

                try:
                    # 获取接口配置
                    from .config_loader import ConfigLoader
                    config_loader = ConfigLoader()
                    interface_config = config_loader.get_interface_config(interface_name)

                    # 处理数据（包含批次内去重）
                    df = self.processor.process_data(task['data'], interface_config)

                    if df.is_empty():
                        logger.warning(f"No data to save after processing: {interface_name}")
                        continue

                    # 验证数据
                    validation_result = self.processor.validate_data(df, interface_config)

                    # 基于接口配置与Parquet文件去重
                    dedup_config = interface_config.get('dedup', {})
                    if dedup_config.get('enabled', False):
                        strategy = dedup_config.get('strategy', 'none')
                        dedup_columns = dedup_config.get('columns', [])

                        if strategy == 'primary_key' and dedup_columns:
                            # 读取现有数据（只读主键列）
                            existing_df = self.read_interface_data(
                                interface_name,
                                columns=dedup_columns
                            )

                            if not existing_df.is_empty():
                                # 构建现有主键集合
                                existing_keys = set()
                                for row in existing_df.iter_rows(named=True):
                                    key_tuple = tuple(row.get(k) for k in dedup_columns if k in row)
                                    if all(v is not None for v in key_tuple):
                                        existing_keys.add(key_tuple)

                                logger.info(f"Found {len(existing_keys)} existing keys for {interface_name}")

                                # 过滤新数据
                                original_count = len(df)
                                new_records = []
                                for row in df.iter_rows(named=True):
                                    key_tuple = tuple(row.get(k) for k in dedup_columns)
                                    if key_tuple not in existing_keys:
                                        new_records.append(row)

                                if not new_records:
                                    logger.info(f"All {original_count} records already exist, skipping")
                                    continue

                                # 重新创建DataFrame
                                df = pl.DataFrame(new_records)
                                logger.info(f"Filtered {original_count - len(df)} duplicates, saving {len(df)} new records")

                    # 保存数据（异步写入）
                    self.save_data(interface_name, df.to_dicts(), async_write=True)

                    logger.info(f"Processed and queued {len(df)} records for {interface_name}")

                except Exception as e:
                    logger.error(f"Error processing {interface_name}: {str(e)}")

                    # 重试机制
                    if task['retry_count'] < 2:
                        task['retry_count'] += 1
                        logger.info(f"Retrying {interface_name}, attempt {task['retry_count'] + 1}/3")
                        self.process_queue.put(task)
                    else:
                        logger.error(f"Failed after 3 retries for {interface_name}")
                        self.failed_interfaces.add(interface_name)

                        # 输出详细的错误信息
                        import traceback
                        traceback.print_exc()

                        # 记录未成功保存的数据情况
                        logger.error(f"Failed to save {len(task['data'])} records for {interface_name}")

            except queue.Empty:
                # 队列为空，检查是否停止
                if not self.running:
                    break
                continue

            except Exception as e:
                logger.error(f"Unexpected error in process worker: {str(e)}")
                if not self.running:
                    break

    def get_failed_interfaces(self) -> Set[str]:
        """获取失败的接口集合"""
        return self.failed_interfaces.copy()

    def clear_failed_interface(self, interface_name: str):
        """清除接口的失败状态"""
        self.failed_interfaces.discard(interface_name)

    def get_buffer_stats(self) -> Dict[str, Any]:
        """获取缓存统计信息"""
        with self.buffer_lock:
            stats = {
                'total_interfaces': len(self.interface_buffers),
                'interface_stats': {}
            }

            for interface_name, buffer in self.interface_buffers.items():
                stats['interface_stats'][interface_name] = {
                    'buffer_count': buffer['count'],
                    'buffer_size_mb': len(str(buffer['data'])) / 1024 / 1024 if buffer['data'] else 0
                }

            return stats
```

```python
def run_concurrent_stock_download(downloader, scheduler, interface_name,
                                  interface_config, base_params, stock_list,
                                  rate_limiter, storage_manager, processor):
    """
    运行并发股票下载 - 监控接口失败状态

    Args:
        downloader: 下载器
        scheduler: 调度器
        interface_name: 接口名称
        interface_config: 接口配置
        base_params: 基础参数
        stock_list: 股票列表
        rate_limiter: 速率限制器
        storage_manager: 存储管理器
        processor: 数据处理器

    Returns:
        总记录数
    """
    logger.info(f"Starting concurrent download for {interface_name} with {len(stock_list)} stocks")

    # 清除接口的失败状态（重新开始）
    storage_manager.clear_failed_interface(interface_name)

    total_records = 0
    tasks = []

    # 创建包装函数
    def download_single_stock_with_rate_limit(interface_config, stock, params):
        rate_limiter.wait_for_tokens(1)

        # 检查接口是否失败
        if interface_name in storage_manager.get_failed_interfaces():
            logger.warning(f"Interface {interface_name} failed, skipping stock {stock['ts_code']}")
            return []

        return downloader.download_single_stock(interface_config, stock, params)

    # 构建任务列表
    for stock in stock_list:
        task = {
            'func': download_single_stock_with_rate_limit,
            'args': (interface_config, stock, base_params),
            'kwargs': {}
        }
        tasks.append(task)

        # 每批提交100个任务
        if len(tasks) >= 100:
            logger.info(f"Submitting batch of {len(tasks)} tasks")

            # 检查接口状态
            if interface_name in storage_manager.get_failed_interfaces():
                logger.error(f"Interface {interface_name} failed, stopping download")
                break

            results = scheduler.submit_tasks(tasks)
            for result in results:
                if result:
                    total_records += len(result)

            logger.info(f"Completed batch, total records: {total_records}")
            tasks = []

    # 提交剩余任务
    if tasks and interface_name not in storage_manager.get_failed_interfaces():
        logger.info(f"Submitting final batch of {len(tasks)} tasks")
        results = scheduler.submit_tasks(tasks)
        for result in results:
            if result:
                total_records += len(result)

    # 检查最终结果
    if interface_name in storage_manager.get_failed_interfaces():
        logger.error(f"Interface {interface_name} failed during download")
        return 0

    logger.info(f"Completed download for {interface_name}, total {total_records} records")
    return total_records
```

## 六、核心算法

### 6.1 缓存累积算法

```python
def add_to_buffer(self, interface_name: str, data: List[Dict]):
    with self.buffer_lock:
        # 1. 初始化或获取缓存
        if interface_name not in self.interface_buffers:
            self.interface_buffers[interface_name] = {'data': [], 'count': 0}

        buffer = self.interface_buffers[interface_name]

        # 2. 累积数据
        buffer['data'].extend(data)
        buffer['count'] += len(data)

        # 3. 检查阈值
        if buffer['count'] >= self.buffer_threshold:
            # 4. 复制数据（避免引用问题）
            data_to_process = buffer['data'].copy()

            # 5. 清空缓存（允许新数据写入）
            buffer['data'] = []
            buffer['count'] = 0

            # 6. 异步处理（不阻塞下载）
            self.process_queue.put({
                'interface_name': interface_name,
                'data': data_to_process,
                'retry_count': 0
            })
```

**关键设计**：
- **锁保护**：确保多线程安全
- **复制数据**：避免后续修改影响处理
- **立即清空**：允许新数据继续累积
- **异步放入队列**：下载线程不阻塞

### 6.2 处理重试算法

```python
def _process_worker(self):
    while True:
        task = self.process_queue.get()

        if task is None:  # 停止信号
            break

        try:
            # 处理数据
            self._process_and_save_data(task)

        except Exception as e:
            # 失败重试逻辑
            if task['retry_count'] < 2:
                task['retry_count'] += 1
                self.process_queue.put(task)  # 重新放入队列
            else:
                # 3次失败，标记接口为失败
                self.failed_interfaces.add(task['interface_name'])
                logger.error(f"Failed after 3 retries: {task['interface_name']}")
```

**关键设计**：
- **重试计数**：task['retry_count']记录重试次数
- **重新入队**：失败任务放回队列末尾
- **失败标记**：3次失败后加入failed_interfaces
- **日志记录**：详细记录错误信息和数据情况

**关键设计**：

## 十、优缺点分析

### 10.1 优点

| 优点 | 说明 |
|------|------|
| **内存速度快** | 数据在内存中累积，无需等待IO |
| **批量处理** | 5000条处理一次，减少小文件 |
| **异步不阻塞** | 下载线程不受处理影响 |
| **自动去重** | 批次内 + 跨批次去重 |
| **失败重试** | 3次重试机制，提高可靠性 |
| **接口隔离** | 每个接口独立缓存和失败状态 |
| **职责清晰** | StorageManager统一管理缓存和队列 |

### 10.2 缺点

| 缺点 | 说明 | 缓解措施 |
|------|------|---------|
| **内存占用** | 多个接口同时缓存，占用内存 | 阈值可调，监控内存 |
| **数据延迟** | 数据累积到5000条才保存 | 程序结束时flush剩余数据 |
| **实现复杂** | 需要处理线程、重试、失败通知 | 封装在StorageManager中 |
| **并发问题** | 多线程访问缓存需要锁 | 使用threading.Lock保护 |

### 10.3 与方案2对比

| 维度 | 方案2（单worker顺序） | 方案3（多worker异步） |
|------|---------------------|---------------------|
| **下载速度** | 较慢（单worker） | **较快**（多worker并发） |
| **内存占用** | 较低（单接口） | **较高**（多接口缓存） |
| **实现复杂度** | 简单（75行） | **中等**（210行） |
| **CPU利用率** | 较低 | **较高**（异步处理） |
| **适用场景** | 小规模数据 | **大规模数据** |
| **健壮性** | 一般 | **高**（重试机制） |

### 13.2 完整代码示例

**完整StorageManager类**（简化版）：
```python
import threading
import queue
import time
from typing import List, Dict, Any, Set
import polars as pl
import logging

logger = logging.getLogger(__name__)

class StorageManager:
    def __init__(self, processor, storage_dir="../data", format="parquet", batch_size=100):
        self.processor = processor
        self.storage_dir = storage_dir
        self.format = format
        self.batch_size = batch_size

        # 写入队列和线程
        self.data_queue = queue.Queue()
        self.writer_thread = None

        # 新增：缓存和处理
        self.interface_buffers = {}
        self.process_queue = queue.Queue()
        self.process_thread = None
        self.buffer_threshold = 5000
        self.buffer_lock = threading.Lock()
        self.failed_interfaces = set()

        self.running = False

    def start_writer(self):
        if not self.running:
            self.running = True
            self.writer_thread = threading.Thread(target=self._writer_worker, daemon=True)
            self.writer_thread.start()
            self.process_thread = threading.Thread(target=self._process_worker, daemon=True)
            self.process_thread.start()
            logger.info("Storage threads started")

    def stop_writer(self):
        if self.running:
            self.running = False
            self.process_queue.put(None)
            if self.process_thread:
                self.process_thread.join()
            self.data_queue.put(None)
            if self.writer_thread:
                self.writer_thread.join()
            logger.info("Storage threads stopped")

    def add_to_buffer(self, interface_name: str, data: List[Dict[str, Any]]):
        with self.buffer_lock:
            if interface_name not in self.interface_buffers:
                self.interface_buffers[interface_name] = {'data': [], 'count': 0}

            buffer = self.interface_buffers[interface_name]
            buffer['data'].extend(data)
            buffer['count'] += len(data)

            if buffer['count'] >= self.buffer_threshold:
                data_to_process = buffer['data'].copy()
                buffer['data'] = []
                buffer['count'] = 0

                try:
                    self.process_queue.put({
                        'interface_name': interface_name,
                        'data': data_to_process,
                        'retry_count': 0
                    }, block=False)
                except queue.Full:
                    logger.error(f"Process queue full for {interface_name}")

    def _process_worker(self):
        while True:
            try:
                task = self.process_queue.get(timeout=1)
                if task is None:
                    break

                interface_name = task['interface_name']
                if interface_name in self.failed_interfaces:
                    continue

                try:
                    from .config_loader import ConfigLoader
                    config_loader = ConfigLoader()
                    interface_config = config_loader.get_interface_config(interface_name)

                    df = self.processor.process_data(task['data'], interface_config)

                    if df.is_empty():
                        continue

                    # 与Parquet文件去重
                    dedup_config = interface_config.get('dedup', {})
                    if dedup_config.get('enabled', False):
                        dedup_columns = dedup_config.get('columns', [])
                        existing_df = self.read_interface_data(interface_name, columns=dedup_columns)

                        if not existing_df.is_empty():
                            existing_keys = set()
                            for row in existing_df.iter_rows(named=True):
                                key = tuple(row.get(k) for k in dedup_columns if k in row)
                                existing_keys.add(key)

                            new_records = []
                            for row in df.iter_rows(named=True):
                                key = tuple(row.get(k) for k in dedup_columns)
                                if key not in existing_keys:
                                    new_records.append(row)

                            if new_records:
                                df = pl.DataFrame(new_records)
                            else:
                                continue

                    self.save_data(interface_name, df.to_dicts(), async_write=True)

                except Exception as e:
                    if task['retry_count'] < 2:
                        task['retry_count'] += 1
                        self.process_queue.put(task)
                    else:
                        self.failed_interfaces.add(interface_name)
                        logger.error(f"Failed after 3 retries: {interface_name}")

            except queue.Empty:
                if not self.running:
                    break

    def get_failed_interfaces(self) -> Set[str]:
        return self.failed_interfaces.copy()

    def clear_failed_interface(self, interface_name: str):
        self.failed_interfaces.discard(interface_name)

    def read_interface_data(self, interface_name: str, columns=None):
```
