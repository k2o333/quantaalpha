```python
def validate_and_adjust_date(start_date, end_date):
    """
    验证并调整日期，确保不使用未来日期
    """
    from datetime import datetime

    today = datetime.now().strftime('%Y%m%d')

    if end_date > today:
        logger.warning(f"结束日期 {end_date} 超过当前日期，已调整为 {today}")
        end_date = today

    if start_date > today:
        logger.warning(f"开始日期 {start_date} 超过当前日期，已调整为 {today}")
        start_date = today

    return start_date, end_date
```

## 步骤5: 添加broker_recommend特殊处理

### 5.1 文件位置
`/home/quan/testdata/aspipe_v4/app4/main.py`

### 5.2 操作步骤

找到第457行附近的普通下载逻辑（在"普通模式，使用同步下载"注释之后），在普通下载逻辑之前插入以下代码：

```python
# 特殊接口处理：broker_recommend需要month参数
if interface_name == 'broker_recommend':
    import pandas as pd
    from datetime import datetime

    # 转换日期范围为月份列表
    start_dt = datetime.strptime(params['start_date'], '%Y%m%d')
    end_dt = datetime.strptime(params['end_date'], '%Y%m%d')

    # 生成月份列表
    months = pd.date_range(
        start_dt,
        end_dt,
        freq='MS'  # 月初
    ).strftime('%Y%m').tolist()

    all_data = []
    for month in months:
        month_params = {'month': month}

        # 如果有ts_code参数，也传递
        if 'ts_code' in params:
            month_params['ts_code'] = params['ts_code']

        # 下载该月数据
        interface_rate_limit = interface_config.get('permissions', {}).get('rate_limit', global_rate_limit)
        interface_rate_limiter = RateLimiter(interface_rate_limit)
        interface_rate_limiter.wait_for_tokens(1)

        month_data = downloader.download(interface_name, month_params)

        if month_data:
            all_data.extend(month_data)
            logger.info(f"Downloaded {len(month_data)} records for {month}")

    if all_data:
        logger.info(f"Successfully downloaded {len(all_data)} total records for {interface_name}")
        process_and_save_data(all_data, interface_name, interface_config, processor, storage_manager)
    else:
        logger.warning(f"No data downloaded for {interface_name}")

    continue  # 跳过普通下载逻辑
```
